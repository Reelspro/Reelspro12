const crypto = require('crypto');
const ENCRYPTION_KEY = process.env.ENCRYPTION_KEY; // must be exactly 32 chars in .env

function encryptApiKey(text) {
  const iv = crypto.randomBytes(16);
  const cipher = crypto.createCipheriv('aes-256-cbc', Buffer.from(ENCRYPTION_KEY), iv);
  let encrypted = cipher.update(text);
  encrypted = Buffer.concat([encrypted, cipher.final()]);
  return iv.toString('hex') + ':' + encrypted.toString('hex');
}

function decryptApiKey(text) {
  try {
    const parts = text.split(':');
    const iv = Buffer.from(parts[0], 'hex');
    const encryptedText = Buffer.from(parts[1], 'hex');
    const decipher = crypto.createDecipheriv('aes-256-cbc', Buffer.from(ENCRYPTION_KEY), iv);
    let decrypted = decipher.update(encryptedText);
    decrypted = Buffer.concat([decrypted, decipher.final()]);
    return decrypted.toString();
  } catch (e) {
    return text; // fallback if not encrypted
  }
}

const aiProviderService = require('./ai_provider_service');

async function generateSuspenseScenes(articleTitle, articleContent, emotion, userId) {
  const prompt = `
    You are a viral suspense reel writer.
    Convert this article into exactly 5 short suspense scenes for a 10-second video reel.

    Article Title: ${articleTitle}
    Article Content: ${(articleContent || articleTitle).substring(0, 1000)}
    Emotion: ${emotion}

    Rules:
    - Each scene: maximum 8 words
    - Scene 1: Shocking hook (grab attention instantly)
    - Scene 2-3: Suspense buildup (short punchy facts)
    - Scene 4: Cliffhanger (leave them wanting more)
    - Scene 5: ALWAYS exactly → "📖 Full Read Story Details In Comments 👇"
    - Tone: dark, cinematic, suspenseful
    - NO full sentences. Short. Punchy. Dramatic.

    Return ONLY valid JSON, no explanation, no markdown:
    {"scenes":[{"id":1,"type":"hook","text":"...","duration":2.0},{"id":2,"type":"beat","text":"...","duration":2.0},{"id":3,"type":"beat","text":"...","duration":2.0},{"id":4,"type":"cliffhanger","text":"...","duration":2.0},{"id":5,"type":"cta","text":"📖 Full Read Story Details In Comments 👇","duration":2.0}]}
  `;

  let result;
  try {
    const provider = await aiProviderService.getAvailableProvider(userId);
    let rawResponse;

    if (provider.provider === 'groq') rawResponse = await aiProviderService.callGroq(prompt, provider.model, provider.apiKey);
    else if (provider.provider === 'gemini') rawResponse = await aiProviderService.callGemini(prompt, provider.model, provider.apiKey);
    else if (provider.provider === 'openrouter') rawResponse = await aiProviderService.callOpenRouter(prompt, provider.model, provider.apiKey);
    else rawResponse = await aiProviderService.callHuggingFace(prompt, provider.model, provider.apiKey);

    const jsonMatch = rawResponse.match(/\{[\s\S]*\}/);
    if (!jsonMatch) throw new Error('No JSON in AI response');

    const parsed = JSON.parse(jsonMatch[0]);
    result = {
      scenes: calculateSceneTiming(parsed.scenes),
      caption: generateCaption(parsed.scenes, emotion, articleTitle),
      hashtags: generateHashtags(emotion),
      cta: generateCTA(),
      aiProvider: provider.provider,
      aiModel: provider.model
    };
  } catch (err) {
    console.error('AI scene generation failed, using fallback:', err.message);
    const fallback = aiProviderService.fallbackToTemplate(articleTitle, emotion);
    result = {
      scenes: fallback.scenes,
      caption: generateCaption(fallback.scenes, emotion, articleTitle),
      hashtags: generateHashtags(emotion),
      cta: generateCTA(),
      aiProvider: 'fallback',
      aiModel: 'template'
    };
  }

  return result;
}

function calculateSceneTiming(scenes) {
  if (!scenes || scenes.length === 0) return scenes;
  const ctaDuration = 2.0;
  const otherScenes = scenes.filter(s => s.type !== 'cta');
  const otherDuration = (10 - ctaDuration) / otherScenes.length;
  return scenes.map(s => ({
    ...s,
    duration: s.type === 'cta' ? ctaDuration : parseFloat(otherDuration.toFixed(2))
  }));
}

const EMOTION_KEYWORDS = {
  horror:       ['murder','dead','kill','blood','death','body','ghost','haunted','terrifying'],
  mystery:      ['missing','vanished','secret','unknown','disappeared','hidden','strange'],
  crime:        ['arrested','police','stolen','robbery','criminal','gang','fraud','scam'],
  emotional:    ['family','mother','child','love','heartbreak','tears','cancer','died'],
  shocking:     ['caught','exposed','leaked','scandal','shocking','unbelievable','viral'],
  suspense:     ['thriller','chase','escape','trap','danger','warning','attack'],
  funny:        ['funny','hilarious','comedy','laugh','prank','joke'],
  motivational: ['success','rich','million','business','inspire','achieve']
};

function detectEmotion(title, content) {
  const text = ((title || '') + ' ' + (content || '').substring(0, 500)).toLowerCase();
  let bestEmotion = 'suspense';
  let bestScore = 0;
  for (const [emotion, keywords] of Object.entries(EMOTION_KEYWORDS)) {
    const score = keywords.filter(kw => text.includes(kw)).length;
    if (score > bestScore) { bestScore = score; bestEmotion = emotion; }
  }
  return bestEmotion;
}

function matchSoundtrack(emotion) {
  const fs = require('fs');
  const path = require('path');
  const db = require('../config/db');

  const tryFolder = (em) => {
    const folder = path.join(__dirname, '../../assets/music', em);
    if (!fs.existsSync(folder)) return null;
    const files = fs.readdirSync(folder).filter(f => f.endsWith('.mp3') || f.endsWith('.wav'));
    if (files.length === 0) return null;
    return { folder, files, emotion: em };
  };

  let found = tryFolder(emotion) || tryFolder('suspense');
  if (!found) return null;

  // Pick file with lowest use_count from music_library
  const tracked = db.prepare(
    "SELECT * FROM music_library WHERE category = ? ORDER BY use_count ASC, RANDOM() LIMIT 1"
  ).get(found.emotion);

  let filename, filePath;
  if (tracked) {
    filename = tracked.filename;
    filePath = tracked.file_path;
  } else {
    filename = found.files[Math.floor(Math.random() * found.files.length)];
    filePath = path.join(found.folder, filename);
  }

  return { filename, filePath, emotion: found.emotion };
}

async function scanMusicLibrary() {
  const fs = require('fs');
  const path = require('path');
  const db = require('../config/db');
  const musicRoot = path.join(__dirname, '../../assets/music');
  if (!fs.existsSync(musicRoot)) return;

  let count = 0;
  const folders = fs.readdirSync(musicRoot);
  for (const category of folders) {
    const folderPath = path.join(musicRoot, category);
    if (!fs.statSync(folderPath).isDirectory()) continue;
    const files = fs.readdirSync(folderPath).filter(f => f.endsWith('.mp3') || f.endsWith('.wav'));
    for (const filename of files) {
      const filePath = path.join(folderPath, filename);
      try {
        db.prepare(`
          INSERT OR IGNORE INTO music_library (filename, category, file_path)
          VALUES (?, ?, ?)
        `).run(filename, category, filePath);
        count++;
      } catch (e) {}
    }
  }
  console.log('Music library scanned: ' + count + ' tracks found');
}

function prepareMusicConfig(filePath, reelDuration) {
  return {
    inputPath: filePath,
    duration: reelDuration,
    fadeIn: 0.5,
    fadeOut: 1.0,
    volume: 0.7,
    audioFilter: `afade=in:d=0.5,afade=out:st=${reelDuration - 1}:d=1,volume=0.7`
  };
}

module.exports = { detectEmotion, matchSoundtrack, scanMusicLibrary, prepareMusicConfig };

async function renderReel({ reelId, scenesJson, musicPath, theme, articleImageUrl, onProgress }) {
  const tempDir = path.join(__dirname, '../../temp', reelId);
  const outputPath = path.join(__dirname, '../../storage/reels', reelId + '.mp4');
  const thumbPath = path.join(__dirname, '../../storage/thumbnails', reelId + '.jpg');

  // Step 1: Create temp dir
  fs.mkdirSync(tempDir, { recursive: true });

  // Step 2: Remotion render frames (--sequence flag required)
  if (onProgress) onProgress(20);
  await execPromise(
    `npx remotion render ReelComposition "${tempDir}/frames" --sequence --props='${JSON.stringify({ scenes: scenesJson, theme, articleImageUrl })}' --frames=0-299`,
    { cwd: path.join(__dirname, '../../../frontend') }
  );

  // Step 3: FFmpeg combine frames + audio → MP4
  if (onProgress) onProgress(50);
  await new Promise((resolve, reject) => {
    const cmd = ffmpeg()
      .input(tempDir + '/frames/%04d.png')
      .inputOptions(['-framerate 30'])
      .outputOptions(['-c:v libx264', '-preset fast', '-crf 23', '-movflags +faststart', '-pix_fmt yuv420p']);

    if (musicPath && fs.existsSync(musicPath)) {
      cmd.input(musicPath)
         .audioCodec('aac')
         .audioBitrate('128k')
         .audioFilters('afade=in:d=0.5,afade=out:st=9:d=1,volume=0.7')
         .outputOptions(['-shortest']);
    }

    cmd.output(outputPath).on('end', resolve).on('error', reject).run();
  });

  // Step 4: Generate thumbnail at 2 second mark
  if (onProgress) onProgress(90);
  await new Promise((resolve, reject) => {
    ffmpeg(outputPath)
      .screenshots({ timestamps: ['00:00:02'], filename: reelId + '.jpg', folder: path.dirname(thumbPath) })
      .on('end', resolve).on('error', reject);
  });

  // Step 5: Cleanup temp frames
  fs.rmSync(tempDir, { recursive: true, force: true });

  if (onProgress) onProgress(100);
  return { outputPath, thumbPath };
}

module.exports = { renderReel };

const { Worker } = require('bullmq');
const { redis } = require('../config/redis');
const renderService = require('../services/render_service');
const { getIO } = require('../services/socket_service');
const db = require('../config/db');

const renderWorker = new Worker('renderQueue', async (job) => {
  const { reelId, userId, scenesJson, musicPath, theme, articleImageUrl } = job.data;

  try {
    db.prepare("UPDATE reels SET status='rendering', render_progress=10 WHERE id=?").run(reelId);
    db.prepare("UPDATE render_jobs SET status='processing', started_at=CURRENT_TIMESTAMP WHERE reel_id=?").run(reelId);
    getIO().emit('render_progress', { reelId, userId, progress: 10, status: 'rendering' });
    job.updateProgress(10);

    const { outputPath, thumbPath } = await renderService.renderReel({
      reelId, scenesJson, musicPath, theme, articleImageUrl,
      onProgress: (pct) => {
        db.prepare("UPDATE reels SET render_progress=? WHERE id=?").run(pct, reelId);
        getIO().emit('render_progress', { reelId, userId, progress: pct, status: 'rendering' });
        job.updateProgress(pct);
      }
    });

    const filePath = '/storage/reels/' + reelId + '.mp4';
    const thumbnailPath = '/storage/thumbnails/' + reelId + '.jpg';
    db.prepare("UPDATE reels SET status='done', file_path=?, thumbnail_path=?, render_progress=100 WHERE id=?")
      .run(filePath, thumbnailPath, reelId);
    db.prepare("UPDATE render_jobs SET status='completed', completed_at=CURRENT_TIMESTAMP, progress=100 WHERE reel_id=?")
      .run(reelId);

    getIO().emit('render_complete', { reelId, userId, downloadUrl: filePath, thumbnailUrl: thumbnailPath });
    job.updateProgress(100);

  } catch (err) {
    db.prepare("UPDATE reels SET status='failed', render_progress=0 WHERE id=?").run(reelId);
    db.prepare("UPDATE render_jobs SET status='failed', error_message=? WHERE reel_id=?").run(err.message, reelId);
    getIO().emit('render_failed', { reelId, userId, error: err.message });
    throw err;
  }

}, {
  connection: redis,
  concurrency: 2,
  defaultJobOptions: { attempts: 3, backoff: { type: 'exponential', delay: 5000 } }
});

module.exports = { renderWorker };

    }

    socket.off('render_complete');
    socket.off('render_failed');
